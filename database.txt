CREATE TABLE stops (
    stop_id varchar PRIMARY KEY,
    name varchar,
    region varchar,
    timestamp timestamptz,
    data jsonb
);

CREATE TABLE ROUTES (
    route_id varchar PRIMARY KEY,
    thread_id varchar,
    name varchar,
    type varchar,
    region varchar,
    timestamp timestamptz,
    data jsonb
);

CREATE TABLE queue (
    id serial PRIMARY KEY,
    type varchar,
    data_id varchar,
    thread_id varchar,
);
